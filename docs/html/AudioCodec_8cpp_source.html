24double AudioCodec::entropy(vector<int> vec){
25 unordered_map<int, double> ch;
26 double entropy_ch=0, prob;
27 int total = vec.size(), value;
28
29 for(int i : vec){
30 value = i;
31 ch[i]++;
32 }
33
34 for(auto it:ch) {
35 prob = double (it.second / double(total));
36 entropy_ch += prob * log2(prob);
37 }
38 return -entropy_ch;
